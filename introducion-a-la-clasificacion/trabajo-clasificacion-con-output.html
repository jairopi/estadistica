m1 <- - thetaopt[2] / (2 * thetaopt[4])

  thetaopt[3]^2 / (4 * thetaopt[5]) -
  thetaopt[1]

r2 <- sqrt(R2 / thetaopt[5])
x1e <- seq(m1 - r1, m1 + r1, length.out = 100)
